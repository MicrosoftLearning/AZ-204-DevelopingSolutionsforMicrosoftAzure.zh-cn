    /*The following lines of code to add using directives for the

    public static class Echo

        [FunctionName("Echo")]
        public static IActionResult Run(        
            [HttpTrigger("POST")] HttpRequest request,
            ILogger logger)

            logger.LogInformation("Received a request");
